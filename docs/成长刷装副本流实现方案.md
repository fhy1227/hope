## 一、现状与目标对比

### 当前流程

```
MainMenu (继续/新游戏)
    │
    ▼
Combat (无限波次生存)
    ├── 战斗 → 商店 → 战斗 → ... 循环
    └── 死亡 → DeathDialog → MainMenu
```

### 目标流程

```
MainMenu
    │
    ▼
Hub (主城)
    ├── 装备/背包管理
    ├── 查看状态(等级/经验/金币)
    ├── 购买消耗品
    └── 选择副本 → 进入
          │
          ▼
    Dungeon Instance (副本)
    ├── 有限波次 (5-8波) + 关底 Boss
    ├── 战斗 → 商店 → 战斗 → ... → Boss战
    ├── 通关 → 结算画面(稀有掉落)
    └── 死亡 → 结算画面(损失部分金币)
          │
          ▼
    Hub (成长)
    ├── 获得经验/升级
    ├── 整理背包/装备
    └── 选择下一个副本
```

### 核心差异

| 维度 | 当前 | 目标 |
|------|------|------|
| **单局时长** | 无限（可一直打） | **约 20 分钟**（固定波次 + Boss） |
| **游戏节奏** | 无尽波次 | **5-8 波小怪 → Boss → 结算** |
| **关卡结构** | 单一竞技场 | **多种副本类型**可选 |
| **通关条件** | 无（只有死亡） | **击败 Boss = 通关** |
| **结算机制** | 死亡 → 回到菜单 | **通关或死亡 → 结算 → 回城** |
| **局外成长** | 仅存档装备/金币 | **等级 + 经验 + 金币 + 装备 + 成就** |
| **失败代价** | 无 | **死亡损失部分金币（非装备）** |

```csharp
public bool IsDungeonUnlocked(DungeonConfig dungeon)
{
    var save = PersistenceMgr.Instance.CurrentSave;
    return save.Level >= dungeon.MinPlayerLevel
        && (dungeon.RequiredClearedDungeonId <= 0
            || save.ClearedDungeons.Contains(dungeon.RequiredClearedDungeonId));
```

```csharp
// WaveManager.cs — 改造
public partial class WaveManager : Node
{
    [Signal] public delegate void BossWaveStartedEventHandler();
    [Signal] public delegate void DungeonCompletedEventHandler();

    private int _totalWaves;
    private int _bossWave;

    public void Initialize(DungeonConfig dungeon)
    {
        _totalWaves = dungeon.TotalWaves;
        _bossWave = dungeon.BossWave;
    }

    protected override void OnWaveTimerEnd()
    {
        if (CurrentWave >= _totalWaves)
        {
            // 所有波次完成 → Boss 战
            EmitSignal(SignalName.BossWaveStarted);
            StartBossWave();
            return;
        }

        // 正常进入商店或下一波
        base.OnWaveTimerEnd();
    }

    private void StartBossWave()
    {
        // 生成 Boss（特殊的刷怪逻辑）
        _spawner.SpawnBoss(_dungeon.BossConfigId);
        // Boss 战无倒计时，Boss 死亡即为通关
    }
}
```

```csharp
// BossAIController.cs — Boss 行为
public partial class BossAIController : Node
{
    [Export] public float PhaseChangeHealthPercent = 0.5f; // 半血转阶段

    private enum BossPhase { Phase1, Phase2, Phase3 }
    private BossPhase _currentPhase = BossPhase.Phase1;
    private HealthComponent _health;

    public override void _Ready()
    {
        _health = GetParent<HealthComponent>();

        // Boss 死亡 → 通关
        _health.Died += OnBossDefeated;
    }

    private void OnBossDefeated()
    {
        // 通知 RunManager 通关
        EventBus.Instance.EmitDungeonCompleted();

    public override void _Process(double delta)
    {
        // 根据生命值切换阶段
        float healthPercent = _health.CurrentHealth / _health.MaxHealth;
        if (healthPercent < PhaseChangeHealthPercent && _currentPhase == BossPhase.Phase1)
        {
            _currentPhase = BossPhase.Phase2;
            EnterPhase2();
        }
    }

    private void EnterPhase2()
    {
        // 增加攻速、召唤小怪等
    }
}
```

### 6.2 结算逻辑

```csharp
// SettlementManager.cs
public partial class SettlementManager : Node
{
    private SettlementResult _result;

    public override void _Ready()
    {
        // 获取本局统计数据
        _result = CollectRunData();

        // 应用结算
        ApplySettlement();

        // 刷新 UI
        RefreshUI();
    }

    private SettlementResult CollectRunData()
    {
        return new SettlementResult
        {
            IsVictory = RunManager.Instance.IsVictory,
            WaveReached = WaveManager.Instance.CurrentWave,
            EnemiesKilled = RunManager.Instance.TotalKills,
            GoldEarned = RunManager.Instance.RunGoldEarned,
            ExpEarned = RunManager.Instance.RunExpEarned,
            ItemsCollected = InventoryManager.Instance.GetRunItems(),
            IsLevelUp = false // 稍后计算
        };
    }

    private void ApplySettlement()
    {
        var save = PersistenceMgr.Instance.CurrentSave;

        // 结算经验
        long totalExp = CalculateExp(_result);
        save.Experience += totalExp;
        _result.ExpEarned = (int)totalExp;

        // 检查是否升级
        int oldLevel = save.Level;
        while (save.Experience >= GetExpForNextLevel(save.Level))
        {
            save.Experience -= GetExpForNextLevel(save.Level);
            save.Level++;
            _result.IsLevelUp = true;
            OnLevelUp(save.Level);
        }

        // 结算金币
        if (_result.IsVictory)
        {
            // 通关：保留全部金币
            // save.Gold 已在游戏中累加
        }
        else
        {
            // 死亡：损失 30% 金币（下限 0）
            int lostGold = Mathf.FloorToInt(save.Gold * 0.3f);
            save.Gold -= lostGold;
            _result.GoldLost = lostGold;
        }

        // 记录通关
        if (_result.IsVictory)
        {
            if (!save.ClearedDungeons.Contains(_dungeon.DungeonId))
                save.ClearedDungeons.Add(_dungeon.DungeonId);
        }

        // 保存
        PersistenceMgr.Instance.FlushSave();
    }

    private void OnLevelUp(int newLevel)
    {
        // 升级奖励：可以选择属性加成（升级三选一）
        // 或自动增加基础属性
    }

    private long GetExpForNextLevel(int level)
    {
        // 读取 exp_level.json 配置
        return ExpLevelConfigManager.Instance.Get(level).ExpRequired;
    }
}

// SettlementResult 结构
public struct SettlementResult
{
    public bool IsVictory;
    public int WaveReached;
    public int EnemiesKilled;
    public int GoldEarned;
    public int GoldLost;
    public int ExpEarned;
    public bool IsLevelUp;
    public List<ItemSaveData> ItemsCollected;
}
```

```csharp
private void OnEnemyKilled(Enemy enemy)
{
    // 基础经验 = 怪物等级 x 10
    int baseExp = enemy.Stats.Level * 10;

    // 精英/Boss 额外加成
```

```csharp
// ExpSystem.cs
public partial class ExpSystem : Node
{
    public static long GetExpForLevel(int level)
    {
        var cfg = ExpLevelConfigManager.Instance.Get(level);
        return cfg?.ExpRequired ?? 999999;
    }

    public static LevelUpReward GetLevelUpReward(int newLevel)
    {
        var cfg = ExpLevelConfigManager.Instance.Get(newLevel);
        return new LevelUpReward
        {

}
```

### 9.3 死亡惩罚设计

| 情景 | 金币损失 | 装备损失 | 备注 |
|------|----------|----------|------|
| **死亡** | 失去本局获得金币的 50% + 已有金币的 10% | 无 | 背包里本局捡的装备保留 |
| **通关** | 保留全部 | 无 | 正常结算 |

```csharp
// 死亡损失公式
private int CalculateDeathPenalty()
{
    var save = PersistenceMgr.Instance.CurrentSave;
    int runGold = RunManager.Instance.RunGoldEarned;    // 本局赚的
    int existingGold = save.Gold - runGold;              // 进本前有的

    // 损失 = 本局赚的 50% + 原有的 10%
    return Mathf.FloorToInt(runGold * 0.5f) + Mathf.FloorToInt(existingGold * 0.1f);
}
```
