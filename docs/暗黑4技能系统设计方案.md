## 一、系统总览

### 1.1 暗黑4 技能系统的核心理念

暗黑4 的技能系统设计遵循以下原则：

```
1. 选择有意义（Meaningful Choices）
   └─ 技能点有限，无法点满所有技能 → 玩家必须取舍

2. 流派差异化（Build Diversity）
   └─ 基础技能选择 + 核心技能选择 + 关键被动 = 不同流派
   └─ 例：旋风斩野蛮人 vs  Hammer of the Ancients 野蛮人

3. 成长感（Progression Feeling）
   └─ 每级 1 个技能点，新的技能不断解锁
   └─ Rank 1→5 有明确数值成长曲线

4. 可视化反馈（Visual Clarity）
   └─ 技能树布局让玩家一目了然看到完整 build 路径
```

### 1.2 核心特性速查表

| 特性 | 说明 | 实现难度 |
|------|------|----------|
| **技能树分页** | 基础→核心→防御→召唤/诡术→终极→关键被动 | ★☆☆☆☆ |
| **技能点数投入** | 每级获得技能点，自由分配 | ★☆☆☆☆ |
| **技能等级强化** | 每技能最多 5 级，每级提升数值 | ★★☆☆☆ |
| **前置条件** | 高阶技能须前置技能已投入点数 | ★★☆☆☆ |
| **强化分支** | 每技能 2 选 1 或 3 选 1 强化 | ★★★☆☆ |
| **关键被动** | 技能树底部大型被动，全流派定调 | ★★★☆☆ |
| **附魔/专精系统** | 职业特有槽位，放入技能获得被动效果 | ★★★★☆ |
| **幸运一击** | 每次命中概率触发额外效果 | ★★★★☆ |
| **易伤/压制** | 独立伤害乘区，核心战斗机制 | ★★★☆☆ |

### 2.2 SkillDefinition（技能定义资源）

这是系统中最重要的数据类，代表一个技能的静态模板。选用 Godot 的 `Resource` 系统，而非硬编码，使技能数据完全可配置。

```csharp
// SkillDefinition.cs — 技能模板资源，可在 Inspector 中编辑
using Godot;
using System;

namespace Hope.SkillSystem;

/// <summary>
/// 技能每级数值表
/// </summary>
[Serializable]
public class SkillRankData
{
    [Export] public int rank;                   // 等级 1~5
    [Export] public float damageMultiplier;     // 伤害倍率（1.0 = 100%）
    [Export] public float cooldownReduction;    // 此等级减少的冷却秒数
    [Export] public float resourceCostModifier; // 资源消耗倍率（1.0 = 原始消耗）
    [Export] public float additionalRange;      // 额外射程（米）
    [Export] public float additionalRadius;     // 额外范围半径（米）
    [Export] public string descriptionOverride; // 覆盖描述（含 {value} 变量插值）
}

/// <summary>
/// 强化分支选项
/// </summary>
[Serializable]
public class SkillEnhancement
{
    [Export] public string enhancementId;       // 唯一 ID，如 "bash_enhance_1"
    [Export] public string displayName;         // 显示名称
    [Export] public string description;         // 效果描述，支持 {value} 插值
    [Export] public string effectModifierId;    // 关联的效果修饰符 ID
    [Export] public Texture2D icon;             // 强化图标（可选）
}

/// <summary>
/// 精华：技能定义资源
/// 创建方式：右键 → New Resource → SkillDefinition
/// 存储路径：res://data/skills/<职业>/<技能名>.tres
/// </summary>
[GlobalClass]
public partial class SkillDefinition : Resource
{
    // ── 基础信息 ──
    [ExportCategory("基础信息")]
    [Export] public string skillId;                 // 唯一 ID，如 "barb_bash"
    [Export] public string skillName;               // 显示名称，如 "猛击"
    [Export] public string description;             // 基础描述，支持 {damage} {rank} 模板变量
    [Export] public string flavorText;              // 背景故事文本（可选）
    [Export] public Texture2D icon;                 // 技能图标
    [Export] public Texture2D iconHighlighted;      // 高亮图标

    // ── 分类 ──
    [ExportCategory("分类")]
    [Export] public ESkillTag tag;                  // 技能标签
    [Export] public ESkillTargetType targetType;    // 目标类型
    [Export] public EDamageType damageType;         // 伤害类型

    // ── 等级 ──
    [ExportCategory("等级")]
    [Export] public int maxRank = 5;                // 最大等级（基础/核心=5，被动=3，终极=1）
    [Export] public int pointsRequired;             // 前置所需总点数
    [Export] public string[] prerequisiteIds;       // 前置技能 ID 数组（空 = 无前置）
    [Export] public SkillRankData[] rankData;       // 每个等级的数值

    // ── 强化 ──
    [ExportCategory("强化分支")]
    [Export] public SkillEnhancement[] enhancements; // 2~3 个强化选项

    // ── 效果数据 ──
    [ExportCategory("效果数据")]
    [Export] public SkillEffectResource effectResource; // 效果资源引用
    [Export] public bool isPassive;                 // 是否为被动技能
    [Export] public bool isUltimate;                // 是否为终极技能

    // ── 视觉 ──
    [ExportCategory("视觉/音效")]
    [Export] public string castAnimation;           // 施法动画名称
    [Export] public string weaponAnimation;         // 武器动画覆盖
    [Export] public string castVfxPath;             // 施法特效效果路径
    [Export] public string hitVfxPath;              // 命中特效路径
    [Export] public string castSfxPath;             // 施法音效路径
    [Export] public string hitSfxPath;              // 命中音效路径

    // ── 运行时 ──
    [ExportCategory("运行时数据（只读）")]
    [Export] public int treePositionX;              // 技能树 X 坐标（编辑器设置）
    [Export] public int treePositionY;              // 技能树 Y 坐标（编辑器设置）

    // ── 辅助方法 ──
    public float GetDamageAtRank(int rank)
    {
        if (rankData == null || rank < 1 || rank > rankData.Length)
            return 0f;
        return rankData[rank - 1].damageMultiplier;
    }

    public string GetDescriptionAtRank(int rank)
    {
        if (rankData == null || rank < 1 || rank > rankData.Length)
            return description;

        var descOverride = rankData[rank - 1].descriptionOverride;
        if (!string.IsNullOrEmpty(descOverride))
            return descOverride.Replace("{rank}", rank.ToString());

        return description
            .Replace("{damage}", (GetDamageAtRank(rank) * 100).ToString("F0"))
            .Replace("{rank}", rank.ToString());
    }
}
```

### 2.3 SkillEffectResource（效果定义资源）

技能的运行时数据模板，与 SkillDefinition 分离，便于复用一个效果模板到多个技能上。

```csharp
// SkillEffectResource.cs — 技能效果资源
using Godot;
using System;

namespace Hope.SkillSystem;

/// <summary>
/// 状态效果定义
/// </summary>
[Serializable]
public class StatusEffectDef
{
    [Export] public string effectId;                // 唯一 ID
    [Export] public string displayName;             // 显示名称
    [Export] public EPassiveTrigger trigger;        // 触发时机
    [Export] public float duration;                 // 持续时间（秒）
    [Export] public float tickInterval;             // 间隔时间（0 = 一次性）
    [Export] public float magnitude;                // 效果强度
    [Export] public float maxStacks;                // 最大叠加层数（1 = 不可叠加）
    [Export] public bool isDebuff;                  // 是否为负面效果
    [Export] public bool removeOnHit;               // 受伤是否移除
    [Export] public Texture2D icon;                 // 状态图标
    [Export] public string vfxPath;                 // 状态特效路径
}

/// <summary>
/// 命中区域形状
/// </summary>
public enum EHitShape
{
    SingleTarget,   // 单体
    Circle,         // 圆形范围
    Cone,           // 扇形
    Rectangle,      // 矩形（直线弹道）
    Arc,            // 弧形斩击
    Projectile,     // 抛射物（自动寻的/弹道）
    Chain,          // 链式弹跳
}

[GlobalClass]
public partial class SkillEffectResource : Resource
{
    // ── 伤害 ──
    [ExportCategory("伤害参数")]
    [Export] public float baseDamage;               // 基础伤害
    [Export] public float damagePerRank;            // 每级额外伤害
    [Export] public float weaponDamagePercent;      // 武器伤害百分比（200 = 200% 武器伤害）
    [Export] public bool useWeaponDamage;           // 是否使用武器伤害计算

    // ── 范围 ──
    [ExportCategory("范围参数")]
    [Export] public EHitShape hitShape;             // 命中形状
    [Export] public float radius;                   // 圆形半径 / 扇形半径
    [Export] public float coneAngle;                // 扇形角度（度）
    [Export] public float range;                    // 最大射程
    [Export] public float width;                    // 矩形宽度
    [Export] public float height;                   // 矩形长度
    [Export] public int maxTargets;                 // 最大命中目标数（0 = 不限）
    [Export] public bool pierce;                    // 是否穿透
    [Export] public int maxBounces;                 // 链式弹跳次数

    // ── 消耗与冷却 ──
    [ExportCategory("消耗与冷却")]
    [Export] public float cooldown;                 // 基础冷却（秒）
    [Export] public float resourceCost;             // 基础资源消耗
    [Export] public float resourceCostPerSecond;    // 引导技能每秒消耗
    [Export] public float castTime;                 // 施法时间（秒）
    [Export] public float channelDuration;          // 引导持续时间（秒）
    [Export] public bool isChanneling;              // 是否为引导技能
    [Export] public bool isInterruptible;           // 施法是否可被打断

    // ── 物理/弹道 ──
    [ExportCategory("弹道参数")]
    [Export] public string projectileScenePath;     // 抛射物场景路径
    [Export] public float projectileSpeed;          // 抛射物速度
    [Export] public float projectileLifetime;       // 抛射物存活时间
    [Export] public bool projectileHoming;          // 是否追踪敌人
    [Export] public float homingStrength;           // 追踪强度（0~1）

    // ── 特效 ──
    [ExportCategory("特效")]
    [Export] public string castEffectPath;          // 施法特效
    [Export] public string hitEffectPath;           // 命中特效
    [Export] public string trailEffectPath;         // 弹道拖尾
    [Export] public string groundEffectPath;        // 地面残留特效
    [Export] public float hitEffectScale;           // 命中特效缩放
    [Export] public float hitEffectDuration;        // 命中特效持续时间

    // ── 状态效果 ──
    [ExportCategory("状态效果")]
    [Export] public StatusEffectDef[] statusEffects; // 施加的状态列表
    [Export] public ECrowdControlType[] crowdControls; // 控制类型
    [Export] public float crowdControlDuration;     // 控制持续时间
    [Export] public float crowdControlChance;       // 控制触发概率（0~1）

    // ── 特殊 ──
    [ExportCategory("特殊机制")]
    [Export] public float luckyHitChance;           // 幸运一击概率（0~1）
    [Export] public float knockbackForce;           // 击退力度
    [Export] public float pullForce;                // 牵引力度
    [Export] public bool isOffHandUsable;           // 是否可副手施放
    [Export] public string[] tags;                  // 自定义标签（如 "projectile", "melee", "ranged"）

    // ── 工具方法 ──
    public float GetFinalDamage(int rank, float weaponDamage)
    {
        float rankBonus = 1f + ((rank - 1) * 0.1f); // 每级 +10%
        if (useWeaponDamage)
            return weaponDamage * (weaponDamagePercent / 100f) * rankBonus;
        return (baseDamage + damagePerRank * (rank - 1)) * rankBonus;
    }

    public float GetCooldown(int rank)
    {
        return Mathf.Max(0.1f, cooldown - (rank - 1) * 0.5f);
    }

    public float GetResourceCost(int rank)
    {
        return resourceCost + (rank - 1) * resourceCost * 0.05f; // 每级 +5% 消耗
    }
}
```

### 2.5 运行时技能状态

```csharp
// PlayerSkillState.cs — 运行时技能状态（存档/读档核心）
using Godot;
using System.Collections.Generic;

namespace Hope.SkillSystem;

/// <summary>
/// 快捷键槽位状态
/// </summary>
public class SkillSlotState
{
    public string skillId;              // 技能 ID，空字符串 = 空槽
    public int assignedSlot;            // 0~5 对应 Q/W/E/R/1/2
    public bool isEnhanced;             // 是否已选择强化

    public bool IsEmpty => string.IsNullOrEmpty(skillId);
}

/// <summary>
/// 玩家技能状态 —— 管理加点、快捷键、强化选择
/// 该数据直接序列化为 JSON 存盘
/// </summary>
public class PlayerSkillState
{
    // ── 加点数据 ──
    public Dictionary<string, int> investedRanks = new();           // skillId → 投入点数
    public Dictionary<string, string> chosenEnhancements = new();   // skillId → chosen enhancementId
    public string chosenKeyPassiveId = "";                          // 选中的关键被动 ID

    // ── 点数 ──
    public int availablePoints = 0;
    public int totalPointsSpent = 0;

    // ── 快捷键 ──
    public SkillSlotState[] skillSlots = new SkillSlotState[6];    // Q/W/E/R/1/2

    public PlayerSkillState()
    {
        for (int i = 0; i < skillSlots.Length; i++)
        {
            skillSlots[i] = new SkillSlotState { assignedSlot = i };
        }
    }

    // ── 加点方法 ──
    public bool CanInvest(string skillId, SkillDefinition def)
    {
        if (availablePoints <= 0) return false;

        int currentRank = investedRanks.GetValueOrDefault(skillId, 0);
        if (currentRank >= def.maxRank) return false;

        // 检查前置条件
        if (def.prerequisiteIds != null)
        {
            foreach (var prereqId in def.prerequisiteIds)
            {
                if (investedRanks.GetValueOrDefault(prereqId, 0) <= 0)
                    return false;
            }
        }

        // 检查所需总点数
        if (totalPointsSpent < def.pointsRequired) return false;

        // 关键被动互斥
        if (def.isUltimate || def.tag == ESkillTag.KeyPassive)
        {
            if (currentRank > 0) return false; // 只能学 1 级
            if (def.isUltimate && HasUltimateInvested()) return false;
        }

        return true;
    }

    public ESkillCastResult Invest(string skillId, SkillDefinition def)
    {
        if (!CanInvest(skillId, def))
            return ESkillCastResult.InsufficientResource;

        int currentRank = investedRanks.GetValueOrDefault(skillId, 0);
        investedRanks[skillId] = currentRank + 1;
        availablePoints--;
        totalPointsSpent++;

        return ESkillCastResult.Success;
    }

    public bool CanRefund(string skillId, SkillDefinition def)
    {
        int currentRank = investedRanks.GetValueOrDefault(skillId, 0);
        if (currentRank <= 0) return false;

        // 检查是否有其他技能依赖此技能
        // （需全局检查，此处略）
        return true;
    }

    public void Refund(string skillId)
    {
        if (!investedRanks.ContainsKey(skillId)) return;
        int currentRank = investedRanks[skillId];
        availablePoints += currentRank;
        totalPointsSpent -= currentRank;
        investedRanks.Remove(skillId);
        chosenEnhancements.Remove(skillId);
    }

    /// <summary>
    /// 重置所有技能点
    /// </summary>
    public void RefundAll()
    {
        availablePoints += totalPointsSpent;
        totalPointsSpent = 0;
        investedRanks.Clear();
        chosenEnhancements.Clear();
        chosenKeyPassiveId = "";

        // 清空快捷键（仅移除已加点技能的绑定，保留空槽）
        for (int i = 0; i < skillSlots.Length; i++)
        {
            var slot = skillSlots[i];
            if (!string.IsNullOrEmpty(slot.skillId) && !investedRanks.ContainsKey(slot.skillId))
            {
                slot.skillId = "";
                slot.isEnhanced = false;
            }
        }
    }

    private bool HasUltimateInvested()
    {
        // 假设终极技能 ID 包含 "ultimate" 或通过其他标记判断
        foreach (var kv in investedRanks)
        {
            var def = SkillDB.GetDefinition(kv.Key);
            if (def != null && def.isUltimate && kv.Value > 0)
                return true;
        }
        return false;
    }
```

```csharp
using Godot;
using System.Collections.Generic;
using System.Linq;

namespace Hope.SkillSystem;

/// <summary>
/// 技能数据库 —— 自动加载所有 SkillDefinition 资源
/// 注册为 Godot Autoload
/// </summary>
public partial class SkillDB : Node
{
    private static SkillDB _instance;
    private Dictionary<string, SkillDefinition> _definitions = new();
    private Dictionary<ESkillTag, List<SkillDefinition>> _byTag = new();

    public override void _Ready()
    {
        _instance = this;
        LoadAllSkills();
    }

    private void LoadAllSkills()
    {
        // 遍历 data/skills/ 目录下的所有 .tres 文件
        var dir = DirAccess.Open("res://data/skills");
        if (dir == null) return;

        var files = new List<string>();
        CollectFiles(dir, "", files);

        foreach (var filePath in files)
        {
            if (!filePath.EndsWith(".tres") && !filePath.EndsWith(".res"))
                continue;

            var def = ResourceLoader.Load<SkillDefinition>(filePath);
            if (def == null || string.IsNullOrEmpty(def.skillId))
                continue;

            Register(def);
        }

        GD.Print($"[SkillDB] 已加载 {_definitions.Count} 个技能定义");
    }

    private void CollectFiles(DirAccess dir, string path, List<string> files)
    {
        dir.ListDirBegin();
        while (true)
        {
            var file = dir.GetNext();
            if (string.IsNullOrEmpty(file)) break;
            if (file == "." || file == "..") continue;

            var fullPath = path + "/" + file;
            if (dir.CurrentIsDir())
            {
                var subDir = DirAccess.Open("res://data/skills" + fullPath);
                if (subDir != null)
                    CollectFiles(subDir, fullPath, files);
            }
            else
            {
                files.Add("res://data/skills" + fullPath);
            }
        }
        dir.ListDirEnd();
    }

    private void Register(SkillDefinition def)
    {
        _definitions[def.skillId] = def;

        if (!_byTag.ContainsKey(def.tag))
            _byTag[def.tag] = new List<SkillDefinition>();
        _byTag[def.tag].Add(def);
    }

    // ── 静态查询 API ──
    public static SkillDefinition GetDefinition(string skillId)
    {
        return _instance?._definitions.GetValueOrDefault(skillId);
    }

    public static List<SkillDefinition> GetSkillsByTag(ESkillTag tag)
    {
        return _instance?._byTag.GetValueOrDefault(tag) ?? new List<SkillDefinition>();
    }

    public static List<SkillDefinition> GetAllSkills()
    {
        return _instance?._definitions.Values.ToList() ?? new List<SkillDefinition>();
    }

    public static bool Exists(string skillId)
    {
        return _instance != null && _instance._definitions.ContainsKey(skillId);
    }
```

```csharp
/// <summary>
/// 全局事件总线 —— 单例，Autoload
/// 使用 Godot 的 Signal 替代 C# 原生事件，便于编辑器调试
/// </summary>
public partial class EventBus : Node
{
    private static EventBus _instance;
    public static EventBus Instance => _instance;

    // Godot Signal 声明
    [Signal] public delegate void SkillCastEventHandler(string skillId, int rank, Vector3 position);
    [Signal] public delegate void SkillHitEventHandler(string skillId, Node2D target, float damage);
    [Signal] public delegate void SkillEndedEventHandler(string skillId);
    [Signal] public delegate void DamageDealtEventHandler(Node2D target, float damage, EDamageType type);
    [Signal] public delegate void DamageTakenEventHandler(Node2D source, float damage, EDamageType type);
    [Signal] public delegate void EnemyKilledEventHandler(Node2D enemy);
    [Signal] public delegate void SkillLearnedEventHandler(string skillId, int newRank);

    // C# Action 回调注册（对性能敏感的场景用）
    private Dictionary<EGameEvent, List<Delegate>> _callbacks = new();

    public override void _EnterTree()
    {
        _instance = this;
    }

    // ── Signal 方式（推荐用于 UI 更新）──
    public static void Emit(EGameEvent evt, params object[] args)
    {
        // 通过 Signal 发射
        switch (evt)
        {
            case EGameEvent.SkillCast:
                Instance?.EmitSignal(SignalName.SkillCast, (string)args[0], (int)args[1], (Vector3)args[2]);
                break;
            case EGameEvent.SkillHit:
                Instance?.EmitSignal(SignalName.SkillHit, (string)args[0], (Node2D)args[1], (float)args[2]);
                break;
            case EGameEvent.SkillLearned:
                Instance?.EmitSignal(SignalName.SkillLearned, (string)args[0], (int)args[1]);
                break;
            // ... 其他事件类似
        }
    }

    // ── C# Action 方式（推荐用于被动效果）──
    public static void Register(EGameEvent evt, Delegate callback)
    {
        if (_instance == null) return;
        if (!_instance._callbacks.ContainsKey(evt))
            _instance._callbacks[evt] = new List<Delegate>();
        _instance._callbacks[evt].Add(callback);
    }

    public static void Unregister(EGameEvent evt, Delegate callback)
    {
        if (_instance == null) return;
        if (_instance._callbacks.TryGetValue(evt, out var list))
            list.Remove(callback);
    }

    public static void InvokeCallbacks(EGameEvent evt, params object[] args)
    {
        if (_instance == null) return;
        if (_instance._callbacks.TryGetValue(evt, out var list))
        {
            foreach (var cb in list)
                cb.DynamicInvoke(args);
        }
    }
}
```

```csharp
            btn.QueueFree();

        foreach (var line in _connectionLines)
            line.QueueFree();

        var skills = SkillDB.GetSkillsByTag(_currentTag);
        if (skills.Count == 0) return;

        // 按树位置排序（先 Y 后 X，从上到下，从左到右）
        skills = skills.OrderBy(s => s.treePositionY).ThenBy(s => s.treePositionX).ToList();

        // 创建技能节点
        foreach (var def in skills)
        {
            var btn = CreateSkillNode(def);
            _nodeButtons[def.skillId] = btn;
        }

        // 创建连线（在前置技能与当前技能之间）
        foreach (var def in skills)
        {
            if (def.prerequisiteIds == null) continue;
            foreach (var prereqId in def.prerequisiteIds)
            {
                if (_nodeButtons.TryGetValue(prereqId, out var fromBtn) &&
                    _nodeButtons.TryGetValue(def.skillId, out var toBtn))
                {
                    CreateConnectionLine(fromBtn, toBtn, def);
                }
            }
        }

    private SkillNodeButton CreateSkillNode(SkillDefinition def)
    {
        var btn = skillNodeScene.Instantiate<SkillNodeButton>();
        _nodeLayer.AddChild(btn);

        // 位置 = 树坐标 × 间距 + 偏移
        btn.Position = new Vector2(
            def.treePositionX * nodeHorizontalSpacing + _scrollOffset.X,
            def.treePositionY * nodeVerticalSpacing + _scrollOffset.Y
        );
        btn.SetSkill(def);
        return btn;
    }

    private void CreateConnectionLine(SkillNodeButton from, SkillNodeButton to, SkillDefinition def)
    {
        var line = new Line2D();
        _connectionLayer.AddChild(line);

        // 计算起点和终点（节点中心）
        var start = from.Position + from.Size / 2;
        var end = to.Position + to.Size / 2;

        // 绘制贝塞尔曲线
        var midY = (start.Y + end.Y) / 2;
        line.Points = new Vector2[]
        {
            start,
            new Vector2(start.X, midY),
            new Vector2(end.X, midY),
            end,
        };

        // 根据技能解锁状态设置颜色
        var playerState = GameState.Instance.PlayerSkillState;
        bool hasPrereq = playerState.investedRanks.GetValueOrDefault(from.SkillId, 0) > 0;
        bool isLearned = playerState.investedRanks.GetValueOrDefault(def.skillId, 0) > 0;

        line.DefaultColor = isLearned ? lineActiveColor
                         : hasPrereq ? lineAvailableColor
                         : lineLockedColor;
        line.Width = 3f;
        line.Antialiased = true;

        _connectionLines.Add(line);
    }

                }
```

```csharp
// SkillBarManager.cs — 快捷键绑定管理
using Godot;
using System.Collections.Generic;

namespace Hope.UI;

public partial class SkillBarManager : Node
{
    [Export] public NodePath[] slotPaths;           // 6 个 SkillSlot 节点的路径

    private SkillSlotUI[] _slots = new SkillSlotUI[6];

    // 按键映射
    private static readonly Key[] SlotKeys = new Key[]
    {
        Key.Q, Key.W, Key.E, Key.R, Key.Key1, Key.Key2
    };

    public override void _Ready()
    {
        for (int i = 0; i < 6; i++)
        {
            _slots[i] = GetNode<SkillSlotUI>(slotPaths[i]);
            _slots[i].SlotIndex = i;
        }

        // 监听加点事件，刷新技能栏
        EventBus.Instance.SkillLearned += OnSkillChanged;
        EventBus.Instance.SkillRefunded += OnSkillChanged;
        EventBus.Instance.SkillTreeReset += OnSkillReset;

        RefreshAllSlots();
    }

    public override void _UnhandledInput(InputEvent @event)
    {
        if (@event is InputEventKey keyEvent && keyEvent.Pressed && !keyEvent.Echo)
        {
            for (int i = 0; i < SlotKeys.Length; i++)
            {
                if (keyEvent.Keycode == SlotKeys[i])
                {
                    TryCastSlot(i);
                    return;
                }
            }
        }
    }

    private void TryCastSlot(int slotIndex)
    {
        if (slotIndex < 0 || slotIndex >= _slots.Length) return;

        var slotState = GameState.Instance.PlayerSkillState.skillSlots[slotIndex];
        if (slotState.IsEmpty) return;

        var def = SkillDB.GetDefinition(slotState.skillId);
        if (def == null) return;

        // 调用技能施放系统
        SkillCastingSystem.Instance.CastSkill(def, slotState);
    }

    /// <summary>
    /// 将技能拖拽至快捷键槽位
    /// </summary>
    public void AssignSkillToSlot(string skillId, int slotIndex)
    {
        if (slotIndex < 0 || slotIndex >= 6) return;

        var state = GameState.Instance.PlayerSkillState;
        int rank = state.investedRanks.GetValueOrDefault(skillId, 0);
        if (rank <= 0) return; // 未学习的技能不能装备

        // 从旧槽移除
        for (int i = 0; i < state.skillSlots.Length; i++)
        {
            if (state.skillSlots[i].skillId == skillId)
                state.skillSlots[i].skillId = "";
        }

        state.skillSlots[slotIndex].skillId = skillId;
        RefreshSlot(slotIndex);
    }

    private void RefreshAllSlots()
    {
        for (int i = 0; i < 6; i++)
            RefreshSlot(i);
    }

    private void RefreshSlot(int index)
    {
        var state = GameState.Instance.PlayerSkillState;
        var slotState = state.skillSlots[index];
        var def = SkillDB.GetDefinition(slotState.skillId);
        _slots[index].SetSkill(def, index);
    }

    private void OnSkillChanged(string skillId, int newRank)
    {
        RefreshAllSlots();
    }

    private void OnSkillReset()
    {
        RefreshAllSlots();
    }
}
```

### 5.1 CooldownManager（冷却管理器）

```csharp
// CooldownManager.cs — 全局冷却管理，Autoload
using Godot;
using System.Collections.Generic;
using System.Linq;

namespace Hope.Core;

public class CooldownEntry
{
    public string skillId;
    public float totalDuration;
    public float remaining;
    public int slotIndex = -1;          // 关联快捷键槽位（用于 UI 刷新）
    public bool isPaused = false;       // 用于冰冻/时间停止
    public float pauseRemaining;        // 暂停时的剩余时间快照
}

public partial class CooldownManager : Node
{
    private static CooldownManager _instance;
    public static CooldownManager Instance => _instance;

    private Dictionary<string, CooldownEntry> _cooldowns = new();
    private List<string> _toRemove = new();    // 待移除列表，避免迭代中修改

    // 冷却缩减属性（来自装备、被动）
    public float GlobalCooldownReduction { get; set; } = 0f;
    public float SpecificCooldownReduction(string skillId) => 0f; // 单技能 CDR

    public override void _EnterTree()
    {
        _instance = this;
    }

    public override void _Process(double delta)
    {
        float dt = (float)delta;
        _toRemove.Clear();

        foreach (var (skillId, entry) in _cooldowns)
        {
            if (entry.isPaused) continue;

            entry.remaining -= dt;
            if (entry.remaining <= 0)
            {
                _toRemove.Add(skillId);
                EventBus.Emit(EGameEvent.CooldownEnded, skillId);
            }
        }

        foreach (var skillId in _toRemove)
            _cooldowns.Remove(skillId);
    }

    /// <summary>
    /// 启动冷却
    /// </summary>
    public void StartCooldown(string skillId, float baseDuration, int slotIndex = -1)
    {
        // 应用冷却缩减
        float finalDuration = Mathf.Max(0.1f,
            baseDuration * (1f - GlobalCooldownReduction));
        finalDuration = Mathf.Max(0.1f,
            finalDuration - SpecificCooldownReduction(skillId));

        _cooldowns[skillId] = new CooldownEntry
        {
            skillId = skillId,
            totalDuration = finalDuration,
            remaining = finalDuration,
            slotIndex = slotIndex,
        };

        EventBus.Emit(EGameEvent.CooldownStarted, skillId, finalDuration);
    }

    /// <summary>
    /// 查询冷却剩余时间
    /// </summary>
    public float GetRemaining(string skillId)
    {
        return _cooldowns.TryGetValue(skillId, out var entry)
            ? Mathf.Max(0, entry.remaining) : 0f;
    }

    /// <summary>
    /// 是否在冷却中
    /// </summary>
    public bool IsOnCooldown(string skillId)
    {
        return _cooldowns.ContainsKey(skillId) && _cooldowns[skillId].remaining > 0;
    }

    /// <summary>
    /// 获取冷却进度（0~1）
    /// </summary>
    public float GetProgress(string skillId)
    {
        if (!_cooldowns.TryGetValue(skillId, out var entry)) return 1f;
        return 1f - (entry.remaining / entry.totalDuration);
    }

    /// <summary>
    /// 强制结束冷却
    /// </summary>
    public void EndCooldown(string skillId)
    {
        _cooldowns.Remove(skillId);
        EventBus.Emit(EGameEvent.CooldownEnded, skillId);
    }

    /// <summary>
    /// 减少冷却时间（被动/装备效果）
    /// </summary>
    public void ReduceCooldown(string skillId, float amount)
    {
        if (_cooldowns.TryGetValue(skillId, out var entry))
            entry.remaining = Mathf.Max(0, entry.remaining - amount);
    }

    /// <summary>
    /// 暂停所有冷却（冰冻/时间停止时）
    /// </summary>
    public void PauseAll()
    {
        foreach (var entry in _cooldowns.Values)
        {
            entry.isPaused = true;
            entry.pauseRemaining = entry.remaining;
        }
    }

    /// <summary>
    /// 恢复所有冷却
    /// </summary>
    public void ResumeAll()
    {
        foreach (var entry in _cooldowns.Values)
            entry.isPaused = false;
    }

    /// <summary>
    /// 计算冷却缩减后的实际冷却时间
    /// </summary>
    public static float CalculateCooldown(float baseCooldown, float cdrPercent, float flatReduction = 0)
    {
        return Mathf.Max(0.1f, baseCooldown * (1f - cdrPercent) - flatReduction);
    }
}
```

```csharp
// ResourceSystem.cs — 各职业资源基类 + 子类实现
using Godot;

namespace Hope.Player;

public enum EResourceType
{
    Fury,       // 野蛮人 怒气
    Mana,       // 法师 法力
    Essence,    // 死灵 魂能
    Energy,     // 游侠 能量
    Spirit,     // 德鲁伊 灵力
    None,       // 无资源
}

/// <summary>
/// 资源系统抽象基类
/// </summary>
public abstract partial class ResourceSystem : Node
{
    [Export] public float current;
    [Export] public float max = 100f;
    [Export] public float regenPerSecond = 0f;
    [Export] public float regenDelay = 3f;          // 消耗后延迟回复（秒）

    public abstract EResourceType ResourceType { get; }
    public event System.Action<float, float> OnResourceChanged;   // current, max

    protected float _timeSinceLastSpend = 0f;

    public override void _Process(double delta)
    {
        float dt = (float)delta;

        if (regenPerSecond > 0)
        {
            _timeSinceLastSpend += dt;
            if (_timeSinceLastSpend >= regenDelay)
                Generate(regenPerSecond * dt);
        }
    }

    public virtual bool CanAfford(float cost)
    {
        return current >= cost;
    }

    public virtual void Spend(float cost)
    {
        if (!CanAfford(cost)) return;
        current -= cost;
        _timeSinceLastSpend = 0f;
        NotifyChanged();
    }

    public virtual void Generate(float amount)
    {
        current = Mathf.Min(max, current + amount);
        if (amount > 0) NotifyChanged();
    }

    public virtual void SetMax(float newMax)
    {
        max = Mathf.Max(1f, newMax);
        current = Mathf.Min(current, max);
        NotifyChanged();
    }

    protected void NotifyChanged()
    {
        OnResourceChanged?.Invoke(current, max);
        EventBus.Emit(EGameEvent.ResourceChanged, current, max, ResourceType.ToString());
    }

    /// <summary>
    /// 资源百分比（用于 UI 条显示）
    /// </summary>
    public float Percent => max > 0 ? current / max : 0f;
}

// ── 野蛮人：怒气系统 ──
// 特点：攻击产生怒气，技能消耗怒气，脱离战斗后缓慢衰减
public partial class FuryResource : ResourceSystem
{
    public override EResourceType ResourceType => EResourceType.Fury;

    [Export] public float furyDecayRate = 2f;       // 脱战后每秒衰减
    [Export] public float decayDelay = 4f;           // 脱战延迟

    private float _timeSinceLastAction = 0f;

    public FuryResource()
    {
        max = 100f;
        current = 0f;
        regenPerSecond = 0f;   // 怒气不自然回复，靠攻击产生
    }

    public override void _Process(double delta)
    {
        base._Process(delta);
        float dt = (float)delta;

        // 怒气衰减
        if (_timeSinceLastAction > decayDelay && current > 0)
        {
            current = Mathf.Max(0, current - furyDecayRate * dt);
            NotifyChanged();
        }
        _timeSinceLastAction += dt;
    }

    public override void Generate(float amount)
    {
        base.Generate(amount);
        _timeSinceLastAction = 0f;
    }

    public override void Spend(float cost)
    {
        base.Spend(cost);
        _timeSinceLastAction = 0f;
    }
}
```

```csharp
// SkillCastingSystem.cs — 核心施放入口
using Godot;

namespace Hope.Core;

public partial class SkillCastingSystem : Node
{
    private static SkillCastingSystem _instance;
    public static SkillCastingSystem Instance => _instance;

    [Export] public Node2D player;
    [Export] public PlayerResourceManager resourceManager;
    [Export] public StatusManager statusManager;       // 角色状态（沉默/眩晕等）

    private SkillEffectResolver _effectResolver;
    private DamageCalculator _damageCalculator;
    private EffectApplicator _effectApplicator;

    public override void _EnterTree()
    {
        _instance = this;
    }

    public override void _Ready()
    {
        _effectResolver = new SkillEffectResolver();
        _damageCalculator = new DamageCalculator();
        _effectApplicator = new EffectApplicator();
    }

    /// <summary>
    /// 施放技能 —— 外部唯一入口
    /// </summary>
    public ESkillCastResult CastSkill(SkillDefinition def, SkillSlotState slotState)
    {
        // 阶段 1：前置检查
        if (CooldownManager.Instance.IsOnCooldown(def.skillId))
            return ESkillCastResult.OnCooldown;

        if (!resourceManager.CanAfford(def.effectResource.resourceCost))
            return ESkillCastResult.InsufficientResource;

        if (statusManager != null)
        {
            if (statusManager.HasStatus(ECrowdControlType.Stun))
                return ESkillCastResult.Stunned;
            if (statusManager.HasStatus(ECrowdControlType.Silence) && !def.isPassive)
                return ESkillCastResult.Silenced;
            if (statusManager.HasStatus(ECrowdControlType.Root))
                return ESkillCastResult.Rooted;
        }

        // 阶段 2：消耗
        resourceManager.Spend(def.effectResource.resourceCost);
        CooldownManager.Instance.StartCooldown(
            def.skillId,
            def.effectResource.cooldown,
            slotState.assignedSlot
        );

        // 阶段 3：发射事件
        EventBus.Emit(EGameEvent.SkillCast, def.skillId,
            GameState.Instance.PlayerSkillState.investedRanks.GetValueOrDefault(def.skillId, 1),
            player?.GlobalPosition ?? Vector3.Zero);

        // 阶段 4：执行效果（异步——等待动画/前摇结束后真正触发）
        ExecuteEffectDelayed(def, slotState);

        return ESkillCastResult.Success;
    }

    private async void ExecuteEffectDelayed(SkillDefinition def, SkillSlotState slotState)
    {
        var effect = def.effectResource;
        int rank = GameState.Instance.PlayerSkillState.investedRanks
            .GetValueOrDefault(def.skillId, 1);

        // 施法前摇
        if (effect.castTime > 0)
        {
            // 播放施法动画
            float timer = effect.castTime;
            while (timer > 0)
            {
                await ToSignal(GetTree().CreateTimer(0.016f), "timeout");
                timer -= 0.016f;
                // 检查是否被打断
                if (effect.isInterruptible && statusManager?.HasStatus(ECrowdControlType.Stun) == true)
                    return;
            }
        }

        // 触发效果解析器
        var targets = _effectResolver.ResolveTargets(def, rank, player);
        foreach (var target in targets)
        {
            float damage = _damageCalculator.Calculate(def, rank, player, target);
            _effectApplicator.Apply(def, rank, target, damage);
        }

        // 引导技能处理
        if (effect.isChanneling)
        {
            await HandleChanneling(def, rank, slotState);
        }
    }

    private async System.Threading.Tasks.Task HandleChanneling(
        SkillDefinition def, int rank, SkillSlotState slotState)
    {
        var effect = def.effectResource;
        float channelTime = effect.channelDuration;
        float interval = 0.5f; // 每 0.5 秒触发一次伤害

        while (channelTime > 0)
        {
            await ToSignal(GetTree().CreateTimer(interval), "timeout");
            channelTime -= interval;

            // 引导中每秒额外消耗资源
            if (effect.resourceCostPerSecond > 0)
            {
                float costPerTick = effect.resourceCostPerSecond * interval;
                if (!resourceManager.CanAfford(costPerTick))
                    break; // 资源不足，提前中断引导
                resourceManager.Spend(costPerTick);
            }

            var targets = _effectResolver.ResolveTargets(def, rank, player);
            foreach (var target in targets)
            {
                float damage = _damageCalculator.Calculate(def, rank, player, target);
                _effectApplicator.Apply(def, rank, target, damage);
            }
        }

        EventBus.Emit(EGameEvent.SkillEnded, def.skillId);
    }
}
```

### 6.3 目标解析器

```csharp
// SkillEffectResolver.cs — 效果目标解析
using Godot;
using System.Collections.Generic;
using System.Linq;

namespace Hope.Core;

public class SkillEffectResolver
{
    /// <summary>
    /// 解析技能目标列表
    /// </summary>
    public List<Node2D> ResolveTargets(SkillDefinition def, int rank, Node2D caster)
    {
        var effect = def.effectResource;
        if (effect == null) return new List<Node2D>();

        return effect.hitShape switch
        {
            EHitShape.SingleTarget => ResolveSingleTarget(caster, effect),
            EHitShape.Circle => ResolveCircleTargets(caster, effect),
            EHitShape.Cone => ResolveConeTargets(caster, effect),
            EHitShape.Rectangle => ResolveRectangleTargets(caster, effect),
            EHitShape.Arc => ResolveArcTargets(caster, effect),
            EHitShape.Projectile => ResolveProjectileTargets(caster, effect),
            EHitShape.Chain => ResolveChainTargets(caster, effect),
            _ => new List<Node2D>(),
        };
    }

    private List<Node2D> ResolveSingleTarget(Node2D caster, SkillEffectResource effect)
    {
        // 鼠标指向目标 / 最近敌人
        var mousePos = caster.GetGlobalMousePosition();
        var allEnemies = GetEnemiesInRange(caster, effect.range);
        return allEnemies
            .OrderBy(e => e.GlobalPosition.DistanceTo(mousePos))
            .Take(1)
            .ToList();
    }

    private List<Node2D> ResolveCircleTargets(Node2D caster, SkillEffectResource effect)
    {
        var center = caster.GetGlobalMousePosition();
        return GetEnemiesInRange(center, effect.radius)
            .Take(effect.maxTargets > 0 ? effect.maxTargets : int.MaxValue)
            .ToList();
    }

    private List<Node2D> ResolveConeTargets(Node2D caster, SkillEffectResource effect)
    {
        var direction = (caster.GetGlobalMousePosition() - caster.GlobalPosition).Normalized();
        var halfAngle = effect.coneAngle * 0.5f;
        var allEnemies = GetEnemiesInRange(caster, effect.range);

        return allEnemies
            .Where(e =>
            {
                var toEnemy = (e.GlobalPosition - caster.GlobalPosition).Normalized();
                var angle = direction.AngleTo(toEnemy);
                return Mathf.Abs(angle) <= Mathf.DegToRad(halfAngle);
            })
            .Take(effect.maxTargets > 0 ? effect.maxTargets : int.MaxValue)
            .ToList();
    }

    private List<Node2D> ResolveRectangleTargets(Node2D caster, SkillEffectResource effect)
    {
        var dir = (caster.GetGlobalMousePosition() - caster.GlobalPosition).Normalized();
        var perp = new Vector2(-dir.Y, dir.X);
        var start = caster.GlobalPosition;

        return GetEnemiesInRange(start, effect.range + effect.height)
            .Where(e =>
            {
                var rel = e.GlobalPosition - start;
                float forward = rel.Dot(dir);
                float side = Mathf.Abs(rel.Dot(perp));
                return forward > 0 && forward <= effect.height && side <= effect.width * 0.5f;
            })
            .Take(effect.maxTargets > 0 ? effect.maxTargets : int.MaxValue)
            .ToList();
    }

    private List<Node2D> ResolveArcTargets(Node2D caster, SkillEffectResource effect)
    {
        // 弧线斩击：前方扇形范围，额外考虑斩击宽度
        return ResolveConeTargets(caster, effect);
    }

    private List<Node2D> ResolveProjectileTargets(Node2D caster, SkillEffectResource effect)
    {
        // 抛射物：生成 Projectile 场景，自动追踪或弹道飞行
        // 碰撞检测由 Projectile 节点自行处理
        // 此处返回空列表，实际目标在 Projectile 命中时确定
        SpawnProjectile(caster, effect);
        return new List<Node2D>();
    }

    private List<Node2D> ResolveChainTargets(Node2D caster, SkillEffectResource effect)
    {
        // 链式弹跳：从鼠标目标开始，依次弹跳到附近敌人
        var mousePos = caster.GetGlobalMousePosition();
        var firstTarget = GetEnemiesInRange(mousePos, 3f)
            .OrderBy(e => e.GlobalPosition.DistanceTo(mousePos))
            .FirstOrDefault();

        if (firstTarget == null) return new List<Node2D>();

        var targets = new List<Node2D> { firstTarget };
        var current = firstTarget;
        var exclude = new HashSet<Node2D> { firstTarget };

        for (int i = 0; i < Mathf.Min(effect.maxBounces, 10); i++)
        {
            var next = GetEnemiesInRange(current, effect.radius + 3f)
                .Where(e => !exclude.Contains(e))
                .OrderBy(e => e.GlobalPosition.DistanceTo(current.GlobalPosition))
                .FirstOrDefault();

            if (next == null) break;
            targets.Add(next);
            exclude.Add(next);
            current = next;
        }

        return targets;
    }

    private List<Node2D> GetEnemiesInRange(Vector2 center, float range)
    {
        // 调用全局敌人查询系统
        // 实际项目中通过空间分区（QuadTree/Grid）优化
        // 这里简化实现
        var enemies = new List<Node2D>();
        // var allEnemies = GameState.Instance.GetAllEnemies();
        // foreach (var e in allEnemies)
        //     if (e.GlobalPosition.DistanceTo(center) <= range)
        //         enemies.Add(e);
        return enemies;
    }

    private List<Node2D> GetEnemiesInRange(Node2D caster, float range)
    {
        return GetEnemiesInRange(caster.GlobalPosition, range);
    }

    private void SpawnProjectile(Node2D caster, SkillEffectResource effect)
    {
        if (string.IsNullOrEmpty(effect.projectileScenePath)) return;
        var scene = ResourceLoader.Load<PackedScene>(effect.projectileScenePath);
        if (scene == null) return;

        var projectile = scene.Instantiate<Projectile>();
        caster.GetParent().AddChild(projectile);
        projectile.GlobalPosition = caster.GlobalPosition;
        projectile.Setup(effect, caster.GetGlobalMousePosition());
    }
}
```

### 6.4 伤害计算器

```csharp
// DamageCalculator.cs — 伤害计算
using Godot;

namespace Hope.Core;

/// <summary>
/// 伤害计算器 —— 实现暗黑4 伤害公式 + 乘法区
/// </summary>
public class DamageCalculator
{
    private static Random _rng = new();

    /// <summary>
    /// 计算技能对单个目标的最终伤害
    /// </summary>
    public float Calculate(SkillDefinition def, int rank, Node2D caster, Node2D target)
    {
        var effect = def.effectResource;
        if (effect == null) return 0f;

        var playerStats = GetPlayerStats(caster);
        var targetStats = GetTargetStats(target);

        // 1. 基础伤害（技能基础值或武器伤害）
        float baseDamage;
        if (effect.useWeaponDamage)
        {
            baseDamage = playerStats.weaponDamage * (effect.weaponDamagePercent / 100f);
        }
        else
        {
            baseDamage = effect.baseDamage + effect.damagePerRank * (rank - 1);
        }

        // 2. 技能等级加成（每级 +10%）
        float rankMultiplier = 1f + (rank - 1) * 0.1f;

        // 3. 主属性加成（核心属性 × 0.1% / 点）
        float mainStatMultiplier = 1f + playerStats.mainStat * 0.001f;

        // 4. 所有增伤之和（从装备/被动获取的 x% 增伤，加法叠加）
        float allDamageBonus = 1f + playerStats.allDamageBonus;

        // 5. 核心伤/基础伤（按技能标签独立乘区）
        float tagDamageMultiplier = 1f;
        switch (def.tag)
        {
            case ESkillTag.Basic:
                tagDamageMultiplier = 1f + playerStats.basicSkillDamageBonus;
                break;
            case ESkillTag.Core:
                tagDamageMultiplier = 1f + playerStats.coreSkillDamageBonus;
                break;
            case ESkillTag.Ultimate:
                tagDamageMultiplier = 1f + playerStats.ultimateSkillDamageBonus;
                break;
        }

        // 6. 暴击判定
        bool isCrit = _rng.NextDouble() < playerStats.critChance;
        float critMultiplier = isCrit
            ? 1f + playerStats.critDamageBonus
            : 1f;

        // 7. 压制判定（强制 3% 基础概率）
        bool isOverpower = _rng.NextDouble() < (playerStats.overpowerChance + 0.03f);
        float overpowerMultiplier = isOverpower
            ? 1f + playerStats.overpowerDamageBonus
            : 1f;

        // 8. 易伤倍率
        float vulnerableMultiplier = targetStats.isVulnerable
            ? 1f + playerStats.vulnerableDamageBonus
            : 1f;

        // 9. 距离增伤
        float distanceMultiplier = 1f;
        float dist = caster.GlobalPosition.DistanceTo(target.GlobalPosition);
        if (dist > 15f) // >15 米算远程
            distanceMultiplier = 1f + playerStats.distantDamageBonus;
        else if (dist < 3f)
            distanceMultiplier = 1f + playerStats.closeDamageBonus;

        // 10. 元素增伤
        float elementalMultiplier = 1f + playerStats.GetElementalDamageBonus(def.damageType);

        // 11. 目标减伤（护甲 + 抗性）
        float targetDamageReduction = targetStats.GetDamageReduction(def.damageType);

        // ── 最终伤害公式 ──
        float finalDamage = baseDamage
                         * rankMultiplier
                         * mainStatMultiplier
                         * allDamageBonus
                         * tagDamageMultiplier
                         * critMultiplier
                         * overpowerMultiplier
                         * vulnerableMultiplier
                         * distanceMultiplier
                         * elementalMultiplier
                         * (1f - targetDamageReduction);

        // 12. 强化分支额外修正
        var state = GameState.Instance.PlayerSkillState;
        if (state.chosenEnhancements.TryGetValue(def.skillId, out var enhId))
        {
            finalDamage *= GetEnhancementDamageModifier(enhId, caster, target, finalDamage);
        }

        // 日志记录
        if (isCrit)
            EventBus.Emit(EGameEvent.CritHit, target, finalDamage);
        if (isOverpower)
            EventBus.Emit(EGameEvent.OverpowerHit, target, finalDamage);

        return Mathf.Max(0, finalDamage);
    }

    /// <summary>
    /// 幸运一击判定
    /// </summary>
    public bool TryLuckyHit(float luckyHitChance, float luckyHitBonus)
    {
        if (luckyHitChance <= 0) return false;
        float finalChance = luckyHitChance * (1f + luckyHitBonus);
        return _rng.NextDouble() < finalChance;
    }

    private PlayerStats GetPlayerStats(Node2D caster)
    {
        // 从 Player 节点获取属性
        if (caster is PlayerController player)
            return player.Stats;
        return new PlayerStats(); // fallback
    }

    private TargetStats GetTargetStats(Node2D target)
    {
        if (target is EnemyController enemy)
            return enemy.Stats;
        return new TargetStats();
    }

    private float GetEnhancementDamageModifier(string enhId, Node2D caster,
        Node2D target, float baseDamage)
    {
        // 根据强化 ID 返回伤害修正
        // 例如：bash_enhance_2 = 双倍伤害 → 返回 2.0
        return 1f;
    }
}
```

### 6.5 效果应用器

```csharp
// EffectApplicator.cs — 将计算好的伤害/效果应用到目标
using Godot;

namespace Hope.Core;

public class EffectApplicator
{
    public void Apply(SkillDefinition def, int rank, Node2D target, float damage)
    {
        var effect = def.effectResource;
        if (effect == null) return;

        // 1. 造成伤害
        if (damage > 0 && target has IDamageable damageable)
        {
            damageable.TakeDamage(damage, def.damageType);

            EventBus.Emit(EGameEvent.DamageDealt, target, damage, def.damageType);
            EventBus.Emit(EGameEvent.SkillHit, def.skillId, target, damage);
        }

        // 2. 施加状态效果
        if (effect.statusEffects != null)
        {
            foreach (var statusDef in effect.statusEffects)
            {
                if (target has IStatusReceiver receiver)
                {
                    // 强化分支检查
                    bool shouldApply = true;
                    var state = GameState.Instance.PlayerSkillState;
                    if (state.chosenEnhancements.TryGetValue(def.skillId, out var enhId))
                    {
                        shouldApply = ShouldStatusApply(enhId, statusDef.effectId);
                    }

                    if (shouldApply)
                    {
                        receiver.ApplyStatus(statusDef);
                        EventBus.Emit(EGameEvent.BuffApplied, statusDef);
                    }
                }
            }
        }

        // 3. 施加控制
        if (effect.crowdControls != null && effect.crowdControls.Length > 0)
        {
            foreach (var cc in effect.crowdControls)
            {
                if (_rng.NextDouble() < effect.crowdControlChance)
                {
                    if (target has ICrowdControllable controllable)
                    {
                        controllable.ApplyCrowdControl(cc, effect.crowdControlDuration);
                        EventBus.Emit(EGameEvent.CrowdControlApplied, target, cc);
                    }
                }
            }
        }

        // 4. 击退/牵引
        if (effect.knockbackForce > 0 && target is Node2D targetNode)
        {
            var dir = (targetNode.GlobalPosition - GetCasterPosition()).Normalized();
            ApplyKnockback(targetNode, dir * effect.knockbackForce);
        }

        // 5. 击杀处理
        if (damage > 0 && target is ICombatEntity entity && entity.Health <= 0)
        {
            EventBus.Emit(EGameEvent.EnemyKilled, target as Node2D);
        }

        // 6. 幸运一击触发检查
        if (effect.luckyHitChance > 0)
        {
            var playerStats = GetPlayerStats();
            if (TryLuckyHit(effect.luckyHitChance, playerStats.luckyHitBonus))
            {
                EventBus.Emit(EGameEvent.LuckyHitProcced, def.skillId, "");
                // 触发装备/被动上的幸运一击效果
            }
        }

        // 7. 视觉反馈
        SpawnHitEffect(target, effect);
    }

    private void ApplyKnockback(Node2D target, Vector2 force)
    {
        if (target is RigidBody2D rb)
        {
            rb.ApplyCentralImpulse(force);
        }
        else if (target is CharacterBody2D character)
        {
            // 对 CharacterBody2D 使用 Tween 动画模拟击退
            var tween = target.CreateTween();
            tween.TweenProperty(target, "position",
                target.Position + force, 0.2f)
                 .SetEase(Tween.EaseType.Out)
                 .SetTrans(Tween.TransitionType.Quad);
        }
    }

    private void SpawnHitEffect(Node2D target, SkillEffectResource effect)
    {
        if (string.IsNullOrEmpty(effect.hitEffectPath)) return;

        var scene = ResourceLoader.Load<PackedScene>(effect.hitEffectPath);
        if (scene == null) return;

        var vfx = scene.Instantiate<Node2D>();
        target.GetParent().AddChild(vfx);
        vfx.GlobalPosition = target.GlobalPosition;
        vfx.Scale = Vector2.One * effect.hitEffectScale;

        // 自动清理
        var timer = new Timer();
        timer.WaitTime = effect.hitEffectDuration > 0 ? effect.hitEffectDuration : 1.0f;
        timer.OneShot = true;
        timer.Timeout += () => { vfx.QueueFree(); timer.QueueFree(); };
        vfx.AddChild(timer);
        timer.Start();
    }

    private Vector2 GetCasterPosition()
    {
        // 简化实现
        return Vector2.Zero;
```

## 七、伤害公式与属性乘法区

### 7.1 暗黑4 伤害乘法区完全解析

暗黑4 的伤害计算采用**独立乘法区**（Multiplicative Damage Buckets）设计。每个区内的来源加法叠加，区与区之间乘法叠加。

```
最终伤害 = 基础伤害 × 区A × 区B × 区C × ... × 区N

各区内部加法（+%），区间乘法（×%）
```

### 7.2 完整乘区表

| 编号 | 乘区名称 | 典型来源 | 示例 |
|------|----------|----------|------|
| **M1** | **基础** | 技能倍率 × 武器伤害 | 猛击 = 130% 武器伤害 |
| **M2** | **主属性** | 1 + (力量/智力/敏捷 × 0.1%) | 500 力量 = 1 + 0.5 = ×1.5 |
| **M3** | **所有伤害** | 所有类型的通用增伤 | +30% 所有伤害 = ×1.3 |
| **M4** | **物理/元素** | 物理/火/冰/毒/暗影增伤 | +20% 火焰伤害 = ×1.2 |
| **M5** | **技能标签** | 基础/核心/终极技能伤害 | +50% 核心技能伤害 = ×1.5 |
| **M6** | **距离** | 近距离/远距离增伤 | 3 米内 +25% = ×1.25 |
| **M7** | **控制** | 对昏迷/减速/冰冻/定身敌人增伤 | +40% 对昏迷敌人 = ×1.4 |
| **M8** | **易伤** | 对易伤敌人增伤 | +20% 易伤伤害 = ×1.2 |
| **M9** | **暴击** | 暴击时，暴击伤害 | 200% 暴击伤害 = ×2.0 |
| **M10** | **压制** | 压制时，压制伤害 | +150% 压制伤害 = ×2.5 |
| **M11** | **精英/首领** | 对精英/首领增伤 | +30% 对精英 = ×1.3 |
| **M12** | **健康/受伤** | 满血/低血额外增伤 | 生命 > 80% 时 +20% = ×1.2 |
| **M13** | **武器精通** | 特定武器类型增伤 | 双手剑 +15% = ×1.15 |
| **M14** | **减伤** | (1 - 目标减伤率) | 护甲 50% 减伤 = ×0.5 |

### 7.3 完整计算公式

```csharp
// 最终伤害计算
public float CalculateFinalDamage(
    SkillDefinition def,
    int rank,
    PlayerStats attacker,
    TargetStats defender)
{
    // ── M1: 基础伤害 ──
    float weaponDamage = attacker.weaponDamage;
    float base = weaponDamage * (def.effectResource.weaponDamagePercent / 100f);
    float rankBonus = 1f + (rank - 1) * 0.1f; // 每级 +10%
    float m1_base = base * rankBonus;

    // ── M2: 主属性 ──
    float m2_mainStat = 1f + attacker.mainStat * 0.001f;

    // ── M3: 所有伤害 ──
    float m3_allDamage = 1f + attacker.allDamageBonus;

    // ── M4: 元素伤害 ──
    float m4_elemental = 1f + attacker.GetElementalDamageBonus(def.damageType);

    // ── M5: 技能标签伤害 ──
    float m5_tagDamage = 1f;
    switch (def.tag)
    {
        case ESkillTag.Basic:   m5_tagDamage = 1f + attacker.basicSkillDamageBonus; break;
        case ESkillTag.Core:    m5_tagDamage = 1f + attacker.coreSkillDamageBonus; break;
        case ESkillTag.Ultimate:m5_tagDamage = 1f + attacker.ultimateSkillDamageBonus; break;
    }

    // ── M6: 距离伤害 ──
    float dist = Vector3.Zero.DistanceTo(defender.position); // 实际需计算
    float m6_distance;
    if (dist < 3f)
        m6_distance = 1f + attacker.closeDamageBonus;
    else if (dist > 15f)
        m6_distance = 1f + attacker.distantDamageBonus;
    else
        m6_distance = 1f;

    // ── M7: 控制伤害 ──
    float m7_cc = defender.HasAnyCrowdControl()
        ? 1f + attacker.ccDamageBonus
        : 1f;

    // ── M8: 易伤 ──
    float m8_vulnerable = defender.isVulnerable
        ? 1f + attacker.vulnerableDamageBonus
        : 1f;

    // ── M9: 暴击 ──
    bool isCrit = Random.Shared.NextDouble() < attacker.critChance;
    float m9_crit = isCrit ? 1f + attacker.critDamageBonus : 1f;

    // ── M10: 压制 ──
    bool isOverpower = Random.Shared.NextDouble() < attacker.overpowerChance;
    float m10_overpower = isOverpower ? 1f + attacker.overpowerDamageBonus : 1f;

    // ── M11: 精英 ──
    float m11_elite = defender.isElite ? 1f + attacker.eliteDamageBonus : 1f;

    // ── M12: 健康/受伤 ──
    float healthPercent = attacker.currentHealth / attacker.maxHealth;
    float m12_health = 1f;
    if (healthPercent > 0.8f) m12_health += attacker.healthyDamageBonus;
    if (healthPercent < 0.35f) m12_health += attacker.injuredDamageBonus;

    // ── M13: 武器精通 ──
    float m13_weapon = 1f + attacker.weaponMasteryBonus;

    // ── M14: 目标减伤 ──
    float m14_mitigation = 1f - defender.GetDamageReduction(def.damageType);

    // ── 最终 ──
    float final = m1_base
                * m2_mainStat
                * m3_allDamage
                * m4_elemental
                * m5_tagDamage
                * m6_distance
                * m7_cc
                * m8_vulnerable
                * m9_crit
                * m10_overpower
                * m11_elite
                * m12_health
                * m13_weapon
                * m14_mitigation;

    return Mathf.Max(0, final);
}
```

### 7.4 护甲与抗性减伤公式

```
护甲减伤 = 护甲 / (护甲 + 敌人等级 × 系数)
        系数 ≈ 1000（具体数值需调参）

抗性减伤 = 抗性% × (1 - 穿透%)
最终减伤 = 1 - (1 - 护甲减伤) × (1 - 抗性减伤)
```

```csharp
public static float CalculateArmorDR(float armor, int enemyLevel)
{
    // 暗黑4 风格的护甲减伤曲线
    return armor / (armor + enemyLevel * 1000f);
}

public static float CalculateResistanceDR(float resistancePercent, float penetrationPercent)
{
    float effectiveResistance = resistancePercent * (1f - penetrationPercent);
    return Mathf.Clamp(effectiveResistance, 0f, 0.85f); // 最大 85% 抗性上限
}
```

### 8.1 易伤（Vulnerable）

```
效果：目标受到的伤害 +20%（基准），持续 4 秒
视觉：目标身上紫色轮廓/破碎效果
叠加：易伤不堆叠，刷新持续时间
```

```csharp
// VulnerableSystem.cs
public partial class VulnerableSystem : Node
{
    private Dictionary<Node2D, float> _vulnerableTargets = new(); // target → remaining time

    public void ApplyVulnerable(Node2D target, float duration, float bonusDamage)
    {
        _vulnerableTargets[target] = duration;

        // 目标身上的易伤标记组件
        if (target is IVulnerableTarget vt)
            vt.SetVulnerable(true, bonusDamage);

        EventBus.Emit(EGameEvent.VulnerableApplied, target, duration);
    }

    public override void _Process(double delta)
    {
        float dt = (float)delta;
        var expired = new List<Node2D>();

        foreach (var (target, remaining) in _vulnerableTargets)
        {
            _vulnerableTargets[target] = remaining - dt;
            if (_vulnerableTargets[target] <= 0)
                expired.Add(target);
        }

        foreach (var target in expired)
        {
            _vulnerableTargets.Remove(target);
            if (target is IVulnerableTarget vt)
                vt.SetVulnerable(false, 0f);
        }
    }
}
```

### 8.2 压制（Overpower）

```
触发条件：
  - 基础概率 3%
  - 技能有压制标签时概率提升
  - 生命/强固状态可增加概率

效果：
  - 伤害 = 基础 × (1 + 压制伤害%)
  - 压制伤害 = 生命值 + 强固值（固定伤害部分）
  - 视觉：屏幕边缘白色闪光，伤害数字加大

特殊：每次施法独立判定，不依赖命中
```

### 8.3 幸运一击（Lucky Hit）

```
机制：
  1. 每个技能有自己的 luckyHitChance（如 20%~50%）
  2. 每次命中时，以该概率判定是否"触发幸运一击"
  3. 触发后，所有装备/被动上的幸运一击效果分别按自己的概率判定

公式：
  最终触发概率 = 技能幸运一击概率 × 效果触发概率

示例：
  技能旋风斩 幸运一击概率 = 25%
  装备"幸运一击时 20% 几率回复怒气"
  → 每击概率 = 25% × 20% = 5%
```

```csharp
// LuckyHitSystem.cs
public partial class LuckyHitSystem : Node
{
    [Export] public PlayerStats playerStats;

    /// <summary>
    /// 检查幸运一击是否触发
    /// </summary>
    public bool TryLuckyHit(string skillId, float skillLuckyHitChance)
    {
        if (skillLuckyHitChance <= 0) return false;
        float finalChance = skillLuckyHitChance * (1f + playerStats.luckyHitBonus);
        return Random.Shared.NextDouble() < finalChance;
    }

    /// <summary>
    /// 检查某个具体效果是否触发（在幸运一击触发后调用）
    /// </summary>
    public bool TryProcEffect(float effectProcChance)
    {
        return Random.Shared.NextDouble() < effectProcChance;
    }
}
```

```csharp
// PassiveSystem.cs — 被动系统
public partial class PassiveSystem : Node
{
    private Dictionary<string, PassiveEffect> _allPassives = new();
    private Dictionary<EPassiveTrigger, List<PassiveEffect>> _triggerMap = new();

    public override void _Ready()
    {
        // 注册所有被动（自动发现 + 手动注册）

        // 监听事件
        EventBus.Instance.SkillHit += OnSkillHit;
        EventBus.Instance.DamageDealt += OnDamageDealt;
        EventBus.Instance.EnemyKilled += OnEnemyKilled;
        EventBus.Instance.CritHit += OnCritHit;
        EventBus.Instance.CooldownEnded += OnCooldownEnded;
    }

    public void RegisterPassive(PassiveEffect passive)
    {
        _allPassives[passive.PassiveId] = passive;

        if (!_triggerMap.ContainsKey(passive.TriggerType))
            _triggerMap[passive.TriggerType] = new();
        _triggerMap[passive.TriggerType].Add(passive);
    }

    /// <summary>
    /// 获取玩家已学习的某个触发类型的被动
    /// </summary>
    private List<(PassiveEffect, int)> GetActivePassives(EPassiveTrigger trigger)
    {
        var result = new List<(PassiveEffect, int)>();
        if (!_triggerMap.TryGetValue(trigger, out var passives)) return result;

        var state = GameState.Instance.PlayerSkillState;
        foreach (var passive in passives)
        {
            int rank = state.investedRanks.GetValueOrDefault(passive.PassiveId, 0);
            if (rank > 0)
                result.Add((passive, rank));
        }
        return result;
    }

    private void ExecutePassives(EPassiveTrigger trigger, PassiveContext ctx)
    {
        var activePassives = GetActivePassives(trigger);
        foreach (var (passive, rank) in activePassives)
            passive.Execute(ctx, rank);
    }

    }

        });
```

## 十一、技能树布局算法

### 11.1 自动布局需求

技能树的节点摆放不是任意的，需要满足：

```
1. 层级清晰 —— 从上到下，难度/强度递增
2. 分支对称 —— 同一层级的多个技能均匀分布
3. 连线不交叉 —— 使用 Bezier 曲线 + 分层路由
4. 滚动可见 —— 超出屏幕的部分可拖拽查看
5. 前置连线 —— 仅绘制直接相连的路径
```

### 11.2 分层布局算法

```csharp
// SkillTreeLayout.cs — 自动布局引擎
public class SkillTreeLayout
{
    /// <summary>
    /// 对给定技能列表进行自动布局
    /// </summary>
    public void AutoLayout(List<SkillDefinition> skills)
    {
        if (skills.Count == 0) return;

        // 1. 拓扑排序（确保前置技能在前）
        var sorted = TopologicalSort(skills);

        // 2. 分配 Y 层（最长为 6 层）
        AssignLayers(sorted);

        // 3. 计算 X 位置（层内居中分布）
        var layers = GroupByLayer(sorted);
        foreach (var (layer, layerSkills) in layers)
        {
            int count = layerSkills.Count;
            int spacing = 120; // 水平间距
            int startX = -(count - 1) * spacing / 2; // 居中
            for (int i = 0; i < count; i++)
            {
                layerSkills[i].treePositionX = startX + i * spacing;
                layerSkills[i].treePositionY = layer;
            }
        }
    }

    private List<SkillDefinition> TopologicalSort(List<SkillDefinition> skills)
    {
        // Kahn 算法拓扑排序
        var graph = new Dictionary<string, List<string>>();
        var inDegree = new Dictionary<string, int>();

        foreach (var s in skills)
        {
            graph[s.skillId] = new List<string>();
            inDegree[s.skillId] = 0;
        }

        foreach (var s in skills)
        {
            if (s.prerequisiteIds == null) continue;
            foreach (var prereq in s.prerequisiteIds)
            {
                if (graph.ContainsKey(prereq))
                {
                    graph[prereq].Add(s.skillId);
                    inDegree[s.skillId]++;
                }
            }
        }

        var queue = new Queue<SkillDefinition>();
        foreach (var s in skills)
        {
            if (inDegree.GetValueOrDefault(s.skillId, 0) == 0)
                queue.Enqueue(s);
        }

        var result = new List<SkillDefinition>();
        while (queue.Count > 0)
        {
            var s = queue.Dequeue();
            result.Add(s);
            foreach (var nextId in graph[s.skillId])
            {
                inDegree[nextId]--;
                if (inDegree[nextId] == 0)
                {
                    var nextDef = skills.Find(x => x.skillId == nextId);
                    if (nextDef != null) queue.Enqueue(nextDef);
                }
            }
        }

        return result;
    }

    private void AssignLayers(List<SkillDefinition> sorted)
    {
        // 层分配：前置技能层 + 1
        foreach (var s in sorted)
        {
            if (s.prerequisiteIds == null || s.prerequisiteIds.Length == 0)
            {
                s.treePositionY = 0;
            }
            else
            {
                int maxPrereqLayer = 0;
                foreach (var prereqId in s.prerequisiteIds)
                {
                    var prereq = SkillDB.GetDefinition(prereqId);
                    if (prereq != null)
                        maxPrereqLayer = Mathf.Max(maxPrereqLayer, prereq.treePositionY);
                }
                s.treePositionY = maxPrereqLayer + 1;
            }
        }
    }

    private Dictionary<int, List<SkillDefinition>> GroupByLayer(List<SkillDefinition> skills)
    {
        var layers = new Dictionary<int, List<SkillDefinition>>();
        foreach (var s in skills)
        {
            if (!layers.ContainsKey(s.treePositionY))
                layers[s.treePositionY] = new List<SkillDefinition>();
            layers[s.treePositionY].Add(s);
        }
        return layers;
    }
}
```

```csharp
// SpatialGrid.cs — 空间网格，优化碰撞检测
public class SpatialGrid
{
    private readonly float _cellSize;
    private readonly Dictionary<(int, int), List<Node2D>> _grid = new();

    public SpatialGrid(float cellSize = 5f)
    {
        _cellSize = cellSize;
    }

    public void Insert(Node2D node)
    {
        var cell = GetCell(node.GlobalPosition);
        if (!_grid.ContainsKey(cell))
            _grid[cell] = new List<Node2D>();
        _grid[cell].Add(node);
    }

    public void Remove(Node2D node)
    {
        var cell = GetCell(node.GlobalPosition);
        if (_grid.TryGetValue(cell, out var nodes))
            nodes.Remove(node);
    }

    public List<Node2D> Query(Vector2 center, float radius)
    {
        var result = new List<Node2D>();
        int cellRadius = Mathf.CeilToInt(radius / _cellSize);
        var centerCell = GetCell(center);

        for (int dx = -cellRadius; dx <= cellRadius; dx++)
        {
            for (int dy = -cellRadius; dy <= cellRadius; dy++)
            {
                var cell = (centerCell.Item1 + dx, centerCell.Item2 + dy);
                if (_grid.TryGetValue(cell, out var nodes))
                {
                    foreach (var node in nodes)
                    {
                        if (node.GlobalPosition.DistanceTo(center) <= radius)
                            result.Add(node);
                    }
                }
            }
        }

        return result;
    }

    private (int, int) GetCell(Vector2 pos)
    {
        return (Mathf.FloorToInt(pos.X / _cellSize),
                Mathf.FloorToInt(pos.Y / _cellSize));
    }
}
```

```csharp
// ObjectPool.cs — 通用对象池
public class ObjectPool<T> where T : Node, new()
{
    private readonly Stack<T> _pool = new();
    private readonly Func<T> _factory;
    private readonly Action<T> _onGet;
    private readonly Action<T> _onReturn;

    public ObjectPool(Func<T> factory, Action<T> onGet = null, Action<T> onReturn = null,
        int prewarmSize = 0)
    {
        _factory = factory;
        _onGet = onGet;
        _onReturn = onReturn;

        for (int i = 0; i < prewarmSize; i++)
        {
            var obj = _factory();
            obj.Visible = false;
            _pool.Push(obj);
        }
    }

    public T Get()
    {
        var obj = _pool.Count > 0 ? _pool.Pop() : _factory();
        _onGet?.Invoke(obj);
        return obj;
    }

    public void Return(T obj)
    {
        _onReturn?.Invoke(obj);
        _pool.Push(obj);
    }
}
```

### 15.2 RPC 接口定义

```csharp
// 多人模式下的技能系统 RPC
public partial class SkillNetworkSync : Node
{
    // 客户端 → 服务端：请求施放
    [Rpc(MultiplayerApi.RpcMode.AnyPeer, CallLocal = false, TransferMode = TransferModeEnum.Reliable)]
    private void RequestCastSkill(int peerId, string skillId, Vector3 targetPos)
    {
        if (!Multiplayer.IsServer()) return;

        var player = GetPlayerByPeerId(peerId);
        if (player == null) return;

        var def = SkillDB.GetDefinition(skillId);
        if (def == null) return;

        // 服务端验证
        var state = player.SkillState;
        if (!state.CanInvest(skillId, def)) return; // 或使用不同的施放验证
        if (CooldownManager.Instance.IsOnCooldown(skillId)) return;

        // 消耗资源
        player.ResourceManager.Spend(def.effectResource.resourceCost);
        CooldownManager.Instance.StartCooldown(skillId, def.effectResource.cooldown);

        // 广播确认给所有客户端
        Rpc(nameof(ConfirmCast), skillId, targetPos, peerId);
    }

    // 服务端 → 所有客户端：确认施放
    [Rpc(MultiplayerApi.RpcMode.Authority, CallLocal = true, TransferMode = TransferModeEnum.Reliable)]
    private void ConfirmCast(string skillId, Vector3 targetPos, int casterPeerId)
    {
        // 在本地施放（已验证过的）
        var def = SkillDB.GetDefinition(skillId);
        SkillCastingSystem.Instance.CastSkillLocal(def, targetPos);
    }

    // 服务端 → 所有客户端：同步冷却
    [Rpc(MultiplayerApi.RpcMode.Authority, TransferMode = TransferModeEnum.Reliable)]
    private void SyncCooldown(string skillId, float duration, int casterPeerId)
    {
        if (Multiplayer.GetUniqueId() == casterPeerId)
            CooldownManager.Instance.StartCooldown(skillId, duration);
    }

    // 服务端 → 所有客户端：命中结果
    [Rpc(MultiplayerApi.RpcMode.Authority, TransferMode = TransferModeEnum.Unreliable)]
    private void SyncHitResult(string skillId, int targetId, float damage, bool isCrit, bool isOverpower)
    {
        // 显示伤害数字、播放特效等
    }

    private PlayerController GetPlayerByPeerId(int peerId) => null;
}
```
